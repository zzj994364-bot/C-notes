统计字符串某个区间内某个字符出现频率

```cpp
#include <iostream>
#include <string>
#include <vector>
using namespace std;
int main() {
ios_base::sync_with_stdio(false);
cin.tie(nullptr);

string s;
cin >> s;
int q;
cin >> q;

// prefix[i] 表示前 i 个字符中 'a' 的数量
vector<int> prefix(s.size() + 1, 0);

for (int i = 0; i < s.size(); i++) {
    // 若当前字符是 'a'，累加到前缀和中
    prefix[i + 1] = prefix[i] + (s[i] == 'a');
}

while (q--) {
    int L, R;
    cin >> L >> R;
    // 区间 [L, R] 的 'a' 数量 = 前R个 - 前(L-1)个
    cout << prefix[R] - prefix[L - 1] << '\n';
}

return 0;
```
