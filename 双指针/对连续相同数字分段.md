## 对连续且相同的数字进行分段

### 1.用vector<pair< >> 管理多个区间

### 2.双指针i，j实现

```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    string s;
    cin >> s; //输入1112222334445556555

    vector<pair<char, int>> v; //相同数字的连续区间
    for (int i = 0; i < s.size();) { // 注意：i 不自增，由内部控制跳跃
        int j = i;
        char c = s[i];
        
        // 找到从 i 开始，这一段相同字符的最右端位置 j
        while (j < s.size() && s[j] == c) j++;
        
        // 现在 [i, j-1] 是一段连续的字符 c
        // 长度为 j - i
        v.push_back({c, j - i});
        
        // 跳到下一段的起点（不要 i++）
        i = j;
    }

}


v = [
    {'1', 3},
    {'2', 4},
    {'3', 2},
    {'4', 3},
    {'5', 3},
    {'6', 1},
    {'5', 3}
];
```
